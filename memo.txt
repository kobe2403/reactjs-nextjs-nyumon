##Dockerfile
FROM node:14.13.1 
WORKDIR /usr/src/app/

##docker-compose.yml
アプリ名：reactapp
ポート：3000

version: '3'
services:
  node:
    build:
      context: .
    tty: true
    environment:
      - NODE_ENV=development
    volumes:
    - ./:/usr/src/app
    command: sh -c "cd reactapp && yarn start"
    ports:
    - "3000:3000"
    tty: true

##typescript版でビルド
docker-compose run --rm node sh -c "npm install -g create-react-app && create-react-app reactapp --template typescript"
